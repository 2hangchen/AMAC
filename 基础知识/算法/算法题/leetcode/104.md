# maximum-depth-of-binary-tree 求树的最大深度
## 解法一
深度优先搜索
```go
func maxDepth(root *TreeNode) int {
    if root == nil {
        return 0
    }
    maxLevel := 0
    dd(root,1,&maxLevel)
    return maxLevel
}
func dd(root *TreeNode,level int, maxLevel *int){
    if root == nil {
        return
    }
    dd(root.Left,level+1,maxLevel)
    dd(root.Right,level+1,maxLevel)
    if *maxLevel <= level {
        *maxLevel = level
    }
}
```
