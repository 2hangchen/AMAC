# topk
## 解法一 堆
```go
type KthLargest struct {
	k       int
	minHeap *IntMinHeap
}

func Constructor(k int, nums []int) KthLargest {
	ktl := KthLargest{minHeap: &IntMinHeap{}, k: k}
	for _, n := range nums {
		heap.Push(ktl.minHeap, n) // 先push
		if ktl.minHeap.Len() > k { // 比k大了再 pop
			heap.Pop(ktl.minHeap)
		}
	}
	return ktl
}

func (ktl *KthLargest) Add(val int) int {
	heap.Push(ktl.minHeap, val) // 先push
	if ktl.minHeap.Len() > ktl.k {  // 当多的数据刚好比k大（1）的时候直接pop然后剩下的数据就是k个，取这k的最小值即可。
		heap.Pop(ktl.minHeap)
	}
	return ktl.minHeap.Peek()
}

type IntMinHeap []int

func (pq *IntMinHeap) Len() int {
	return len(*pq)
}
func (pq *IntMinHeap) Less(i, j int) bool {
	return (*pq)[i] < (*pq)[j]
}
func (pq *IntMinHeap) Swap(i, j int) {
	(*pq)[i], (*pq)[j] = (*pq)[j], (*pq)[i]
}

func (pq *IntMinHeap) Push(x interface{}) {
	*pq = append(*pq, x.(int))
}

func (pq *IntMinHeap) Pop() interface{} {
	n := len(*pq) - 1
	x := (*pq)[n]
	*pq = (*pq)[:n]
	return x
}
func (pq *IntMinHeap) Peek() int {
	return (*pq)[0]
}
```
