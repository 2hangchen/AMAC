func main() {
	tt := [5]int{1,2,3,4,5}
	d := Constructor(3,tt[:])
	fmt.Println(d.Add(6))
	fmt.Println(d.Add(7))
	fmt.Println(d.Add(8))
	fmt.Println(d.Add(9))
	fmt.Println(d.Add(10))
	fmt.Println(d.Add(11))

type KthLargest struct {
	nums []int
	topK []int
}

// k nums
func Constructor(k int, nums []int) KthLargest {
	return KthLargest{
		nums: nums,
		topK: make([]int, k, k),
	}
}

func (k *KthLargest) Add(val int) int {
	v := k.topK
	var a godata.MinHeap
	a = v

	for _,v := range k.nums {
		if v > godata.TopMin(&a) {
			godata.PopMin(&a)
			godata.PushMin(&a,v)
		}
	}
	if val > godata.TopMin(&a) {
		godata.PopMin(&a)
		godata.PushMin(&a,val)
	}

	return godata.PopMin(&a)
}
