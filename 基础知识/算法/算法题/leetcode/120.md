```go
func minimumTotal(triangle [][]int) int {
    result :=  int(^uint(0) >> 1) // 给它一个最大值。
    ma(0,0,0,&result,triangle)
    return result   
}

func ma(i,j,re int,result *int,trangle [][]int){
    //fmt.Println(*result)
    // stop
    if i == len(trangle)-1 {
        re += trangle[i][j]
        if *result > re {
            *result= re
        }
        return
    }
    // process
    value := trangle[i][j]
    // trill down
    ma(i+1,j,re+value,result,trangle)
    ma(i+1,j+1,re+value,result ,trangle)
    //clear states
}
```
