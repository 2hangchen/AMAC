#  Valid Parentheses 有效的括号
## 解法一
```go
func isValid(s string) bool {
	// 设置一个辅助的map
	aid := make(map[string]string)
	aid["("], aid["{"], aid["["] = ")", "}", "]"
	// 设置一个辅助的stack
	stack := godata.NewStack() //github.com/googege/godata  这里使用了一个stack
	for _, v := range s {
		value := string(v)
		if _, ok := aid[value]; ok { // 存在也就是左边
			stack.Push(value)
		} else { //不存在也就是右边
			top := stack.Pop()
			if top != nil {
				if value != aid[top.(string)] {
					return false
				}
			}else {
				return false
			}
		}
	}
	if stack.Length() > 0 {
		return false
	}
	return true
}
```
