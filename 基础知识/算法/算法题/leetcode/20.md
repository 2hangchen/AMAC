#  Valid Parentheses 有效的括号
## 解法一
```go
func isValid(s string) bool {
	// 设置一个辅助的map
	aid := make(map[rune]rune)
	aid['('], aid['{'], aid['['] = ')','}',']'
	// 设置一个辅助的stack
	stack := godata.NewStack(6) //github.com/googege/godata  
	for _, v := range s {
		if _, ok := aid[value]; ok { // 存在也就是左边
			stack.Push(value)
		} else { //不存在也就是右边
			top := stack.Pop()
			if top != nil { // 右边多
				if value != aid[top.(rune)] {
					return false
				}
			}else {
				return false
			}
		}
	}
	if stack.Length() > 0 {
 		return false // 左边多
	}
	return true
}
```
