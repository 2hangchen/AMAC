# 用栈实现队列
## 解法一
```go
type Queue struct {
	stackIn *godata.Stack
	stackOut *godata.Stack
}
func NewQueue()*Queue {
	return &Queue{
		stackIn:  godata.NewStack(10),
		stackOut: godata.NewStack(10),
	}
}
func(q *Queue)Push(v int){
	q.stackIn.Push(v)
}
func(q*Queue)Pop()int{
	if q.stackOut.Length() == 0 {
		for q.stackIn.Length() > 0 {
			value := q.stackIn.Pop()
			q.stackOut.Push(value)
		}
	}
	return q.stackOut.Pop().(int)

}
func(q *Queue)Top()int{
	for q.stackIn.Length() >0 {
		value := q.stackIn.Pop()
		q.stackOut.Push(value)
	}
	return q.stackOut.Top().(int)
}

func(q *Queue)Empty()bool {
	if q.stackOut.Length() ==0 && q.stackIn.Length() == 0 {
		return false
	}
	return true
}

```
