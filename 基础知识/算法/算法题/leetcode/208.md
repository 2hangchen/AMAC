# implement trie prefix tree
## 方法一
```go
type Trie struct { // 字典树使用hash不如数组好使。还有 一个节点的值存在于他的children的k上 所以其实字典树没有节点 所谓值
  // 都是存在于children中的
 Children [26]*Trie
 IsWord bool   
}


/** Initialize your data structure here. */
func Constructor() Trie {
    return Trie{
        Children: [26]*Trie{},
        IsWord : false,
    }
}


/** Inserts a word into the trie. */
func (this *Trie) Insert(word string)  {
    for _,v := range word {
        if this.Children[v-'a'] == nil { // 使用 v - 'a' 是为了获取0值
            this.Children[v-'a'] = &Trie { /// 插入的时候看是否存在这个k了，如果有了就往下遍历，如果没有就新建。
                IsWord:false,
                Children: [26]*Trie{},
            }
        }
        this = this.Children[v-'a'] // 往下遍历 类似 node= node.next
    }
    this.IsWord = true // 将已经 文字结束设置为true
}


/** Returns if the word is in the trie. */
func (this *Trie) Search(word string) bool {

    for _,v := range word {
        if this.Children[v-'a'] == nil {
            return false
        }
        this = this.Children[v-'a']
    }
    return this.IsWord
}


/** Returns if there is any word in the trie that starts with the given prefix. */
func (this *Trie) StartsWith(prefix string) bool {

    for _,v := range prefix {
        if this.Children[v-'a'] == nil  {
            return false
        }
        this = this.Children[v-'a']
    }
    if !this.IsWord { // 这里就是判断前缀是否是跟要找的数据一样长，如果不是再说下面的查找。
        for _,v := range this.Children {
            if v != nil {
                return true
            }
        }
    }
    return this.IsWord
}


/**
 * Your Trie object will be instantiated and called as such:
 * obj := Constructor();
 * obj.Insert(word);
 * param_2 := obj.Search(word);
 * param_3 := obj.StartsWith(prefix);
 */
```

优化办法 ：我们可以将这个数组储存的字符设置为 有序的数组，这样我们就不需要遍历来查找来，我们就可以二分查找了。
