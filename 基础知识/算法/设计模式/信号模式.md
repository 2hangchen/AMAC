# 信号模式
信号模式是一种同步模式 ，将互斥强加给有限数量的资源。

```go
func main() {
	a := Newa()
	a.Send()
	a.Recive()
}

type A interface {
	Send()
	Recive()
}

type B struct {
	ti    time.Duration
	value chan struct{}
}

func (b *B) Send() {
	select {
	case b.value <- struct{}{}:
	case <-time.After(b.ti):
		fmt.Println("over")
	}
}
func (b *B) Recive() {
	select {
	case <-b.value:
		fmt.Println("收到")
	case <-time.After(b.ti):
		fmt.Println("over")
	}
}

func Newa() A {
	return &B{
		ti:    time.Second * 2,
		value: make(chan struct{}, 5),
	}
}

```
